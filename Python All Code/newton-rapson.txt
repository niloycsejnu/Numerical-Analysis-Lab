import math 
def newton_raphson(f, f1, p0, tol, n):
    i = 1
    while i <= n:
        p = p0 - (f(p0)/f1(p0))
        print(format(i,'<4'), p)
        if abs(p-p0) < tol:
            print('The solution is: ',p) 
            return None
        i = i+1
        p0 = p
    print('Method fails after {} iteration'.format(n))
    return None
