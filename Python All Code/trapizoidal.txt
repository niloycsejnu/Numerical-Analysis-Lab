from math import sin, pi
 
def trapezoidal(f, a, b, n):
    h = float(b-a)/n
    y = [0 for i in range(n+1)]
    y[0] = f(a)
    y[n] = f(b)
    result = h*(y[0]+y[n])/2
    for i in range(1, n):
        y[i] = f(a + i*h)
        result = result + h*y[i]
    return result
 
 
f = lambda x: sin(x)
n = 1000000000000000
ans = trapezoidal(f, 0, pi/2, n)
print(ans)
